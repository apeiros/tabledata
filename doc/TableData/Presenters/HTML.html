def html_table_header
  if @table.headers?
    "    <table>\n" +
    "      <thead>\n        <tr>\n"+
      @table.headers.map { |cell|"          <th>#{CGI.escapeHTML(cell)}</th>" }.join("\n")+
      "\n        </tr>\n      </thead>\n"
  else
    "    <table>\n"
  end
end

def table_html(table)
  html_table_header+
    "      <tbody>\n"+
    table.body.map { |row|
      "        <tr>\n"+row.map { |cell| "          <td>#{CGI.escapeHTML(cell)}</td>" }.join("\n")+"\n        </tr>\n"
    }.join("")+
    "      </tbody>\n"+
    "    </table>\n"
end
