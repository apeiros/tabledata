def csv_options
  options = ::CSV::DEFAULT_OPTIONS.dup
  @options.each do |k,v| options[OptionMapping.fetch(k,k)] = v end

  options
end

def string(options=nil)
  ::CSV.generate(csv_options) do |csv|

  end

def write(path, options=nil)
